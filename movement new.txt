-- Load Library
local lb = loadstring(game:HttpGet("https://raw.githubusercontent.com/NormalZyoherion/testyyy/refs/heads/main/green%20libary"))()

local win = lb.new("MOVEMENT")
local mainTab = win:addTab("Main")
local bypassTab = win:addTab("Bypasses")
win:switchTab(mainTab)

-- Services
local Players = game:GetService("Players")
local RunService = game:GetService("RunService")
local UserInputService = game:GetService("UserInputService")
local Workspace = game:GetService("Workspace")

local player = Players.LocalPlayer
local character = player.Character or player.CharacterAdded:Wait()
local humanoid = character:WaitForChild("Humanoid")
local hrp = character:WaitForChild("HumanoidRootPart")

-- ============================================
-- TAB: MAIN (Normal WalkSpeed & JumpPower)
-- ============================================

-- Variables
local normalWalkSpeed = 16
local normalJumpPower = 50
local customWalkSpeed = 50
local customJumpPower = 100
local walkSpeedEnabled = false
local jumpPowerEnabled = false

-- Apply Functions
local function applyWalkSpeed()
    if humanoid then
        humanoid.WalkSpeed = walkSpeedEnabled and customWalkSpeed or normalWalkSpeed
    end
end

local function applyJumpPower()
    if humanoid then
        humanoid.JumpPower = jumpPowerEnabled and customJumpPower or normalJumpPower
        -- Also try JumpHeight for newer games
        humanoid.JumpHeight = jumpPowerEnabled and (customJumpPower / 50 * 7.2) or 7.2
    end
end

-- Anti Reset Loops
task.spawn(function()
    while true do
        task.wait(0.1)
        if walkSpeedEnabled and humanoid and humanoid.WalkSpeed ~= customWalkSpeed then
            applyWalkSpeed()
        end
        if jumpPowerEnabled and humanoid and humanoid.JumpPower ~= customJumpPower then
            applyJumpPower()
        end
    end
end)

-- Respawn Handler
player.CharacterAdded:Connect(function(newChar)
    character = newChar
    humanoid = character:WaitForChild("Humanoid")
    hrp = character:WaitForChild("HumanoidRootPart")
    task.wait(0.2)
    applyWalkSpeed()
    applyJumpPower()
end)

-- Main Tab UI
win:addSection("WalkSpeed & JumpPower")

win:addToggle("Walkspeed:", false, function(state)
    walkSpeedEnabled = state
    applyWalkSpeed()
end)

win:addSlider("WalkSpeed Value", 16, 200, 50, function(v)
    customWalkSpeed = v
    if walkSpeedEnabled then applyWalkSpeed() end
end)

win:addToggle("JumpPower:", false, function(state)
    jumpPowerEnabled = state
    applyJumpPower()
end)

win:addSlider("JumpPower Value", 50, 200, 100, function(v)
    customJumpPower = v
    if jumpPowerEnabled then applyJumpPower() end
end)

-- ============================================
-- TAB: BYPASSES (Anti-Cheat Bypass Methods)
-- ============================================
win:switchTab(bypassTab)

-- CFrame Speed Variables
local cframeWalkSpeed = 5
local cframeRunSpeed = 20
local cframeCurrentSpeed = cframeWalkSpeed
local cframeRunning = false
local cframeEnabled = false

-- Gravity Variables
local defaultGravity = Workspace.Gravity
local customGravity = 50
local gravityEnabled = false

-- JumpPower Bypass Variables (MULTIPLE METHODS)
local jumpBypassEnabled = false
local jumpPowerBypassValue = 100

-- Method 1: Velocity Jump (Instant Upward Velocity)
-- Method 2: CFrame Jump (Teleport Up)
-- Method 3: Platform Stand + Velocity

-- CFrame Speed System
UserInputService.InputBegan:Connect(function(input, gpe)
    if gpe then return end
    if not cframeEnabled then return end
    if input.KeyCode == Enum.KeyCode.LeftShift then
        cframeRunning = true
        cframeCurrentSpeed = cframeRunSpeed
    end
end)

UserInputService.InputEnded:Connect(function(input, gpe)
    if input.KeyCode == Enum.KeyCode.LeftShift then
        cframeRunning = false
        cframeCurrentSpeed = cframeWalkSpeed
    end
end)

RunService.RenderStepped:Connect(function(dt)
    if not cframeEnabled then return end
    if not humanoid or not hrp then return end
    
    local moveDir = humanoid.MoveDirection
    if moveDir.Magnitude > 0 then
        hrp.CFrame = hrp.CFrame + (moveDir * cframeCurrentSpeed * dt)
    end
end)

-- Gravity Bypass
RunService.Heartbeat:Connect(function()
    if gravityEnabled then
        Workspace.Gravity = customGravity
    else
        Workspace.Gravity = defaultGravity
    end
end)

-- ============================================
-- JUMPPOWER BYPASS SYSTEM (MULTI-METHOD)
-- ============================================

-- Method 1: Velocity-Based Jump (Most Reliable)
local function velocityJump()
    if not hrp then return end
    -- Apply instant upward velocity
    hrp.AssemblyLinearVelocity = Vector3.new(
        hrp.AssemblyLinearVelocity.X,
        jumpPowerBypassValue * 0.8,
        hrp.AssemblyLinearVelocity.Z
    )
end

-- Method 2: CFrame-Based Jump (For Games Blocking Velocity)
local function cframeJump()
    if not hrp then return end
    -- Small teleport upward
    hrp.CFrame = hrp.CFrame * CFrame.new(0, jumpPowerBypassValue * 0.05, 0)
end

-- Method 3: BodyVelocity Method (Legacy Style)
local function bodyVelocityJump()
    if not hrp then return end
    local bv = Instance.new("BodyVelocity")
    bv.Velocity = Vector3.new(0, jumpPowerBypassValue, 0)
    bv.MaxForce = Vector3.new(0, math.huge, 0)
    bv.Parent = hrp
    game:GetService("Debris"):AddItem(bv, 0.1)
end

-- Combined Jump Function (Tries All Methods)
local function bypassJump()
    if not jumpBypassEnabled then return end
    if not humanoid or humanoid:GetState() == Enum.HumanoidStateType.Jumping then return end
    
    -- Try all methods for maximum compatibility
    pcall(velocityJump)
    task.wait(0.01)
    pcall(cframeJump)
    task.wait(0.01)
    pcall(bodyVelocityJump)
end

-- Jump Detection (Space Key)
UserInputService.InputBegan:Connect(function(input, gpe)
    if gpe then return end
    if not jumpBypassEnabled then return end
    if input.KeyCode == Enum.KeyCode.Space then
        bypassJump()
    end
end)

-- Alternative: JumpRequest Detection (More Reliable)
UserInputService.JumpRequest:Connect(function()
    if not jumpBypassEnabled then return end
    bypassJump()
end)

-- Auto-Jump Loop (Hold Space)
local autoJumpEnabled = false
RunService.RenderStepped:Connect(function()
    if not jumpBypassEnabled or not autoJumpEnabled then return end
    
    if UserInputService:IsKeyDown(Enum.KeyCode.Space) then
        if humanoid and humanoid:GetState() ~= Enum.HumanoidStateType.Jumping then
            bypassJump()
        end
    end
end)
