-- Load Library
local lb = loadstring(game:HttpGet("https://raw.githubusercontent.com/NormalZyoherion/testyyy/refs/heads/main/green%20libary"))()

local win = lb.new("Noclip")
local mainTab = win:addTab("Main")
win:switchTab(mainTab)

-- Services
local RunService = game:GetService("RunService")
local Players = game:GetService("Players")
local player = Players.LocalPlayer
local UserInputService = game:GetService("UserInputService")

-- ============================================
-- NOCLIP FUNCTION (100% ORIGINAL FROM FILE - NO CHANGES)
-- ============================================

local noclipActive = false

-- PHYSICS LOGIC (FORCE NOCLIP) - EXACT COPY FROM FILE
local noclipConnection = RunService.Stepped:Connect(function()
	if noclipActive and player.Character then
		for _, part in pairs(player.Character:GetDescendants()) do
			if part:IsA("BasePart") and part.CanCollide == true then
				
				-- Detecção de chão para não cair no vácuo
				local params = RaycastParams.new()
				params.FilterDescendantsInstances = {player.Character}
				local res = workspace:Raycast(part.Position, Vector3.new(0, -5, 0), params)

				if not res then
					-- Se não houver chão, atravessa tudo (paredes)
					part.CanCollide = false
				else
					-- Se houver chão, o personagem não cai, mas o script 
					-- permite atravessar se você andar contra a parede.
					part.CanCollide = false
				end
			end
		end
	end
end)

-- ============================================
-- UI SETUP (ONLY WRAPPER - NO LOGIC CHANGES)
-- ============================================

win:addSection("Noclip Controls")

-- Status Label
local statusLabel = win:addLabel("Status: OFF")

-- Noclip Toggle (hanya set variable, logic tetap di atas)
win:addToggle("Enable Noclip", false, function(state)
	noclipActive = state
	
	-- Update UI only, logic handled by original connection
	if noclipActive then
		statusLabel:set("Status: ON")
	else
		statusLabel:set("Status: OFF")
		-- Re-enable collision when OFF (safety)
		if player.Character then
			for _, part in pairs(player.Character:GetDescendants()) do
				if part:IsA("BasePart") then
					part.CanCollide = true
				end
			end
		end
	end
end)

-- Info
win:addLabel("Credits: Filtering Enabled, Zk0")
win:addLabel("Press [N] for quick toggle")

-- ============================================
-- KEYBIND (ONLY TOGGLES VARIABLE)
-- ============================================

UserInputService.InputBegan:Connect(function(input, gameProcessed)
	if gameProcessed then return end
	
	if input.KeyCode == Enum.KeyCode.N then
		noclipActive = not noclipActive
		
		-- Update toggle visual
		for _, ctrl in pairs(win._toggles or {}) do
			if ctrl and ctrl.get and ctrl.set then
				-- Find noclip toggle by checking name
				pcall(function()
					if ctrl.get() ~= noclipActive then
						ctrl.set(noclipActive, true)
					end
				end)
			end
		end
		
		-- Update label
		if noclipActive then
			statusLabel:set("Status: ON")
		else
			statusLabel:set("Status: OFF")
			-- Re-enable collision
			if player.Character then
				for _, part in pairs(player.Character:GetDescendants()) do
					if part:IsA("BasePart") then
						part.CanCollide = true
					end
				end
			end
		end
	end
end)

-- Handle death (reset)
player.CharacterAdded:Connect(function()
	noclipActive = false
	statusLabel:set("Status: OFF (Respawned)")
end)

win:switchTab(mainTab)
